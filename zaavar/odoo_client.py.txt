def get_employee_list():
    """Odoo-оос ажилчдын жагсаалт татах функц"""
    # Эхлээд нэвтэрч Session ID авна
    session_url = "https://erp.erchmining.mn/web/session/authenticate"
    payload = {
        "jsonrpc": "2.0",
        "method": "call",
        "params": {"db": DB, "login": USERNAME, "password": PASSWORD}
    }
    
    try:
        session = requests.Session()
        res = session.post(session_url, json=payload, verify=False)
        
        # Ажилчдын дата татах (Odoo JSON-RPC call)
        search_url = "https://erp.erchmining.mn/web/dataset/call_kw"
        search_payload = {
            "jsonrpc": "2.0",
            "method": "call",
            "params": {
                "model": "hr.employee",
                "method": "search_read",
                "args": [[]], # Бүх ажилчдыг авна
                "kwargs": {
                    "fields": ["id", "name", "work_email", "department_id"],
                    "context": {}
                }
            }
        }
        
        response = session.post(search_url, json=search_payload, verify=False)
        final_result = response.json()
        
        if "result" in final_result:
            return final_result["result"]
        else:
            return []
            
    except Exception as e:
        print(f"Дата татахад алдаа гарлаа: {e}")
        return []
